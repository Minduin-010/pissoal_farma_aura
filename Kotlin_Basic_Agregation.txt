SELECT 
    DATE_FORMAT(trans_date, '%Y-%m') AS month,
    country,
    COUNT(id) AS trans_count,
    SUM(CASE WHEN state = 'approved' THEN 1 ELSE 0 END) AS approved_count,
    SUM(amount) AS trans_total_amount,
    SUM(CASE WHEN state = 'approved' THEN amount ELSE 0 END) AS approved_total_amount
FROM 
    Transactions
GROUP BY 
    DATE_FORMAT(trans_date, '%Y-%m'), 
    country;

Atv_2

SELECT 
    ROUND(
        SUM(CASE WHEN order_date = customer_pref_delivery_date THEN 1 ELSE 0 END) * 100.0 / COUNT(*), 
        2
    ) AS immediate_percentage
FROM 
    Delivery
WHERE 
    (customer_id, order_date) IN (
        SELECT customer_id, MIN(order_date)
        FROM Delivery
        GROUP BY customer_id
    );

WITH RankedOrders AS (
    SELECT 
        order_date,
        customer_pref_delivery_date,
        ROW_NUMBER() OVER (PARTITION BY customer_id ORDER BY order_date) AS rn
    FROM 
        Delivery
)
SELECT 
    ROUND(
        SUM(CASE WHEN order_date = customer_pref_delivery_date THEN 1 ELSE 0 END) * 100.0 / COUNT(*), 
        2
    ) AS immediate_percentage
FROM 
    RankedOrders
WHERE 
    rn = 1;

Atv_3

WITH FirstLogin AS (
    SELECT 
        player_id, 
        MIN(event_date) AS first_date
    FROM 
        Activity
    GROUP BY 
        player_id
)
SELECT 
    ROUND(
        COUNT(A.player_id) * 1.0 / COUNT(F.player_id), 
        2
    ) AS fraction
FROM 
    FirstLogin F
LEFT JOIN 
    Activity A 
ON 
    F.player_id = A.player_id 
    AND A.event_date = DATE_ADD(F.first_date, INTERVAL 1 DAY);

WITH RankedActivity AS (
    SELECT 
        player_id,
        event_date,
        ROW_NUMBER() OVER(PARTITION BY player_id ORDER BY event_date) AS rn,
        LEAD(event_date) OVER(PARTITION BY player_id ORDER BY event_date) AS next_date
    FROM 
        Activity
)
SELECT 
    ROUND(
        SUM(CASE WHEN next_date = DATE_ADD(event_date, INTERVAL 1 DAY) THEN 1 ELSE 0 END) * 1.0 
        / COUNT(*), 
        2
    ) AS fraction
FROM 
    RankedActivity
WHERE 
    rn = 1;
